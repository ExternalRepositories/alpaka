44  //! The CPU memory buffer.
45  template<
46  typename TElem,
47  typename TDim,
48  typename TIdx>
49  class BufCpuImpl final :
50  public alloc::AllocCpuAligned<std::integral_constant<std::size_t, core::vectorization::defaultAlignment>>
51  {
52  static_assert(
53  !std::is_const<TElem>::value,
54  "The elem type of the buffer can not be const because the C++ Standard forbids containers of const elements!");
55  static_assert(
56  !std::is_const<TIdx>::value,
57  "The idx type of the buffer can not be const!");
58  public:
59  //-----------------------------------------------------------------------------
60  template<
61  typename TExtent>
62  ALPAKA_FN_HOST BufCpuImpl(
63  DevCpu const & dev,
64  TExtent const & extent) :
65  alloc::AllocCpuAligned<std::integral_constant<std::size_t, core::vectorization::defaultAlignment>>(),
66  m_dev(dev),
67  m_extentElements(extent::getExtentVecEnd<TDim>(extent)),
68  m_pMem(alloc::alloc<TElem>(*this, static_cast<std::size_t>(computeElementCount(extent)))),
69  m_pitchBytes(static_cast<TIdx>(extent::getWidth(extent) * static_cast<TIdx>(sizeof(TElem))))
70 #if (defined(ALPAKA_ACC_GPU_CUDA_ENABLED) && BOOST_LANG_CUDA) || (defined(ALPAKA_ACC_GPU_HIP_ENABLED) && BOOST_LANG_HIP)
71  ,m_bPinned(false)
72 #endif
73  {
74  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
75 
76  static_assert(
77  TDim::value == Dim<TExtent>::value,
78  "The dimensionality of TExtent and the dimensionality of the TDim template parameter have to be identical!");
79  static_assert(
80  std::is_same<TIdx, Idx<TExtent>>::value,
81  "The idx type of TExtent and the TIdx template parameter have to be identical!");
82 
83 #if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
84  std::cout << __func__
85  << " e: " << m_extentElements
86  << " ptr: " << static_cast<void *>(m_pMem)
87  << " pitch: " << m_pitchBytes
88  << std::endl;
89 #endif
90  }
91  //-----------------------------------------------------------------------------
92  BufCpuImpl(BufCpuImpl const &) = delete;
93  //-----------------------------------------------------------------------------
94  BufCpuImpl(BufCpuImpl &&) = default;
95  //-----------------------------------------------------------------------------
96  auto operator=(BufCpuImpl const &) -> BufCpuImpl & = delete;
97  //-----------------------------------------------------------------------------
98  auto operator=(BufCpuImpl &&) -> BufCpuImpl & = default;
99  //-----------------------------------------------------------------------------
100  ALPAKA_FN_HOST ~BufCpuImpl()
101  {
102  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
103 
104 #if (defined(ALPAKA_ACC_GPU_CUDA_ENABLED) && BOOST_LANG_CUDA) || (defined(ALPAKA_ACC_GPU_HIP_ENABLED) && BOOST_LANG_HIP)
105  // Unpin this memory if it is currently pinned.
106  buf::unpin(*this);
107 #endif
108  // NOTE: m_pMem is allowed to be a nullptr here.
109  alloc::free(*this, m_pMem);
110  }
111 
112  private:
113  //-----------------------------------------------------------------------------
114  //! \return The number of elements to allocate.
115  template<
116  typename TExtent>
117  ALPAKA_FN_HOST static auto computeElementCount(
118  TExtent const & extent)
119  -> TIdx
120  {
121  auto const extentElementCount(extent::getExtentProduct(extent));
122 
123  return extentElementCount;
124  }
125 
126  public:
127  DevCpu const m_dev;
128  Vec<TDim, TIdx> const m_extentElements;
129  TElem * const m_pMem;
130  TIdx const m_pitchBytes;
131 #if (defined(ALPAKA_ACC_GPU_CUDA_ENABLED) && BOOST_LANG_CUDA) || (defined(ALPAKA_ACC_GPU_HIP_ENABLED) && BOOST_LANG_HIP)
132  bool m_bPinned;
133 #endif
134  };
135  }

428  //! The BufCpu memory pinning trait specialization.
429  template<
430  typename TElem,
431  typename TDim,
432  typename TIdx>
433  struct Pin<
434  buf::BufCpu<TElem, TDim, TIdx>>
435  {
436  //-----------------------------------------------------------------------------
437  ALPAKA_FN_HOST static auto pin(
438  buf::BufCpu<TElem, TDim, TIdx> & buf)
439  -> void
440  {
441  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
442 
443  if(!buf::isPinned(buf))
444  {
445 #if (defined(ALPAKA_ACC_GPU_CUDA_ENABLED) && BOOST_LANG_CUDA) || (defined(ALPAKA_ACC_GPU_HIP_ENABLED) && BOOST_LANG_HIP)
446  if(buf.m_spBufCpuImpl->m_extentElements.prod() != 0)
447  {
448  // - cudaHostRegisterDefault:
449  // See http://cgi.cs.indiana.edu/~nhusted/dokuwiki/doku.php?id=programming:cudaperformance1
450  // - cudaHostRegisterPortable:
451  // The memory returned by this call will be considered as pinned memory by all CUDA contexts, not just the one that performed the allocation.
452  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK_IGNORE(
453  ALPAKA_API_PREFIX(HostRegister)(
454  const_cast<void *>(reinterpret_cast<void const *>(view::getPtrNative(buf))),
455  extent::getExtentProduct(buf) * sizeof(Elem<buf::BufCpu<TElem, TDim, TIdx>>),
456  ALPAKA_API_PREFIX(HostRegisterDefault)),
457  ALPAKA_API_PREFIX(ErrorHostMemoryAlreadyRegistered));
458 
459  buf.m_spBufCpuImpl->m_bPinned = true;
460  }
461 #else
462  static_assert(
463  meta::DependentFalseType<TElem>::value,
464  "Memory pinning of BufCpu is not implemented when CUDA or HIP is not enabled!");
465 #endif
466  }
467  }
468  };
